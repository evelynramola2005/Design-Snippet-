    <script>
        document.addEventListener("DOMContentLoaded", function () {
            function createDroplet() {
                const droplet = document.createElement("div");
                droplet.classList.add("droplet");
        
                let size = Math.random() * 8 + 5; // Random size between 5px - 13px
                droplet.style.width = `${size}px`;
                droplet.style.height = `${size}px`;
        
                // Assign random rainbow colors
                const colors = ["#ff0000", "#ff8c00", "#ffd700", "#008000", "#0000ff", "#4b0082", "#9400d3"];
                let randomColor = colors[Math.floor(Math.random() * colors.length)];
                droplet.style.background = randomColor;
                droplet.style.boxShadow = `0 0 10px ${randomColor}`;
        
                // Position droplets along the rainbow
                const rainbow = document.querySelector(".rainbow");
                const rect = rainbow.getBoundingClientRect(); // Get rainbow position
                const rainbowLeft = rect.left; // Left boundary
                const rainbowTop = rect.top; // Top boundary
                const rainbowWidth = rect.width; // Width of the rainbow
        
                let randomX = rainbowLeft + Math.random() * rainbowWidth; // Random point along the rainbow base
                droplet.style.left = `${randomX}px`;
                droplet.style.top = `${rainbowTop + rect.height / 2}px`; // Start falling from the rainbow
        
                // Append to body
                document.body.appendChild(droplet);
        
                // Remove after animation
                setTimeout(() => {
                    droplet.remove();
                }, 4000);
            }
        
            // Generate droplets at an interval
            setInterval(createDroplet, 400);
        });
    </script>

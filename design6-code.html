            <script>
                function randomText() {
                    let text = "abcdefghijklmnopqrstuvwxyz0123456789!@#$%^&";
                    return text[Math.floor(Math.random() * text.length)];
                }
        
                function rain() {
                    let cloud = document.querySelector('.cloud');
                    let e = document.createElement('div');
                    let left = Math.floor(Math.random() * 320);
                    e.classList.add("text");
                    e.innerText = randomText();
                    e.style.position = "absolute"; 
                    e.style.left = left + 'px';
                    e.style.fontSize = (0.5 + Math.random()) + 'em';
                    e.style.top = "0px"; 
                    e.style.color = "white";
        
                    cloud.appendChild(e);
        
                    let fallInterval = setInterval(function () {
                        let topPosition = parseInt(e.style.top) || 0;
                        if (topPosition < 350) { // Stops at the surface
                            e.style.top = topPosition + 5 + 'px';
                        } else {
                            clearInterval(fallInterval);
                            cloud.removeChild(e); // Removes the letter when it falls
                        }
                    }, 30); // Increased speed for heavy rain
                }
        
                setInterval(rain, 100); // Increased frequency for heavier rain
            </script>
        </body>
